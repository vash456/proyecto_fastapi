# iniciar el proyecto
python -m venv app-fastapi

# instalar las librerias de fastapi
pip install fastapi
pip install "fastapi[standard]"    # instalacion mas completa

# luego en la ruta del proyecto crear el archivo requirements.txt
pip freeze > requirements.txt

# crear archivo main.py por fuera de la carpeta del entorno virtual
# para ejecutar ubicarse en el directorio del archivo main.py
fastapi dev

# para ver la documentacion de la api con swagger
http://127.0.0.1:8000/docs

# comando para ver que .exe busca python para ejecutarse
cmd /c where python
